# Technical Analyst agents.
#

# ── 7. Authentication ─────────────────────────────────────────────────────────

# JWT token creation and verification for FastAPI auth endpoints
# Used in Phase 5 (T-046) for self-issued access tokens
python-jose[cryptography]==3.3.0

# Password hashing for self-hosted register/login (T-046)
passlib[bcrypt]==1.7.4

# passlib==1.7.4 (the last released passlib version; upstream is stalled)
# reads bcrypt.__about__.__version__ internally to detect the bcrypt
# backend version. That attribute was removed in bcrypt==4.1.0, which
# only produces a harmless "(trapped) error reading bcrypt version"
# warning on every hash/verify call -- noisy but functionally correct.
# bcrypt>=5.0.0 goes further and breaks passlib's 72-byte truncation
# handling outright, raising "ValueError: password cannot be longer
# than 72 bytes" on ordinary passwords (github.com/pyca/bcrypt/issues/1079).
# Pin to the last version that still ships __about__, exactly like the
# pydyf pin below for the analogous WeasyPrint/pydyf break.
bcrypt==4.0.1


# ── 8. PDF Generation ─────────────────────────────────────────────────────────

# Generates the final Investment Memo PDF from HTML/CSS template
# Used in Phase 4 (T-044) when the Portfolio Manager produces the memo
weasyprint==62.3

# weasyprint==62.3 does not upper-bound its own pydyf dependency, so a bare
# "pip install weasyprint==62.3" can silently pull in a too-new pydyf and
# break PDF rendering with "AttributeError: 'super' object has no attribute
# 'transform'" inside weasyprint/pdf/stream.py (pydyf 0.11.0, released
# 2024-07-12, made a breaking API change -- see
# github.com/Kozea/WeasyPrint/issues/2620). Pin pydyf explicitly below that
# breaking release so this can never silently regress on a fresh install.
pydyf==0.10.0
